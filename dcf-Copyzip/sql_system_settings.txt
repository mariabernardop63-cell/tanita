-- 1. CRIAR TABELA DE CONFIGURAÇÕES
CREATE TABLE IF NOT EXISTS system_settings (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    key TEXT UNIQUE NOT NULL,
    value TEXT,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. ATIVAR SEGURANÇA (Row Level Security)
ALTER TABLE system_settings ENABLE ROW LEVEL SECURITY;

-- 3. Criar POLICIES (permissões)
-- Todos podem ler
DROP POLICY IF EXISTS "public_read_settings" ON system_settings;
CREATE POLICY "public_read_settings" ON system_settings FOR SELECT USING (true);

-- Todos podem atualizar (para o admin poder mudar)
DROP POLICY IF EXISTS "public_update_settings" ON system_settings;
CREATE POLICY "public_update_settings" ON system_settings FOR UPDATE USING (true);

-- Todos podem inserir
DROP POLICY IF EXISTS "public_insert_settings" ON system_settings;
CREATE POLICY "public_insert_settings" ON system_settings FOR INSERT WITH CHECK (true);
