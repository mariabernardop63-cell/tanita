-- Criar a tabela withdrawal_requests
CREATE TABLE IF NOT EXISTS withdrawal_requests (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id TEXT NOT NULL,
    user_name TEXT,
    user_phone TEXT,
    wallet_type TEXT,
    wallet_number TEXT,
    amount DECIMAL(12,2) NOT NULL,
    fee DECIMAL(12,2) DEFAULT 5.00,
    total DECIMAL(12,2) NOT NULL,
    status TEXT DEFAULT 'pending',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Ativar seguranca por linha
ALTER TABLE withdrawal_requests ENABLE ROW LEVEL SECURITY;

-- Policy - admin pode ver todos, users veem os seus proprios
DROP POLICY IF EXISTS "Anyone can read withdrawal_requests" ON withdrawal_requests;
CREATE POLICY "Anyone can read withdrawal_requests" ON withdrawal_requests FOR SELECT USING (true);

-- Policy - qualquer um pode inserir
DROP POLICY IF EXISTS "Anyone can insert withdrawal_requests" ON withdrawal_requests;
CREATE POLICY "Anyone can insert withdrawal_requests" ON withdrawal_requests FOR INSERT WITH CHECK (true);

-- Policy - apenas admin pode atualizar
DROP POLICY IF EXISTS "Admin can update withdrawal_requests" ON withdrawal_requests;
CREATE POLICY "Admin can update withdrawal_requests" ON withdrawal_requests FOR UPDATE USING (auth.role() = 'authenticated');

-- Adicionar indice para melhor performance
CREATE INDEX IF NOT EXISTS idx_withdrawal_requests_status ON withdrawal_requests(status);
CREATE INDEX IF NOT EXISTS idx_withdrawal_requests_created ON withdrawal_requests(created_at DESC);

-- ============================================
-- ATIVAR REALTIME (IMPORTANTÍSSIMO!)
-- ============================================
ALTER PUBLICATION supabase_realtime ADD TABLE withdrawal_requests;
